library(ebmr.alpha)
## 
## Attaching package: 'ebmr.alpha'
## The following object is masked from 'package:stats':
## 
##     fitted.values
set.seed(100)
sd = 1
n = 1000
p = 100
X = matrix(rnorm(n*p),ncol=p)
btrue = rnorm(p)
y = X %*% btrue + sd*rnorm(n)

plot(X %*% btrue, y)

fit = ebmr(X,y)
plot(fitted.values(fit),y)
